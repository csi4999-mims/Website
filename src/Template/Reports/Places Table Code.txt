CREATE TABLE `cake`.`Places` ( `PlacesID` INT NOT NULL AUTO_INCREMENT , `Name` VARCHAR(256) NOT NULL , `Street` VARCHAR(256) NOT NULL , `City` VARCHAR(256) NOT NULL , `State` VARCHAR(256) NOT NULL , `Zip` INT(5) NOT NULL , `Email` VARCHAR(256) NOT NULL , `MiscInfo` TEXT NOT NULL , PRIMARY KEY (`PlacesID`)) ENGINE = InnoDB;
